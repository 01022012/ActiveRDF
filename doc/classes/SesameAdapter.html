  • :inferencing => true or false, if sesame2 rdfs inferencing is uses (defaults to true)

184:     results = []
185:     
186:     # translate the query object into a SPARQL query string
187:     qs = Query2SPARQL.translate(query)
188:     
189:     # evaluate the query on the sesame triple store
190:     # TODO: if we want to get inferred statements back we have to say so, as third boolean parameter
191:     tuplequeryresult = @db.evaluateTupleQuery(QueryLanguage::SPARQL, qs)
192: 
193:     # what are the variables of the query ?
194:     variables = tuplequeryresult.getBindingNames
195:     sizeOfVariables = variables.size
196: 
197:     # a solution is a binding of a variable to all entities that matched this variable in the sparql query
198:     solutionIterator = tuplequeryresult.iterator
199:     
200:     # the following is plainly ugly. the reason is that JRuby currently does not support
201:     # using iterators in the ruby way: with "each". it is possible to define "each" for java.util.Iterator
202:     # using JavaUtilities.extend_proxy but that fails in strange ways. this is ugly but works. 
203:     
204:     # TODO: null handling, if a value is null...
205:     
206:     # if there only was one variable, then the results array should look like this: 
207:     # results = [ [first Value For The Variable], [second Value], ...]
208:     if sizeOfVariables == 1 then
209:       # the counter keeps track of the number of values, so we can insert them into the results at the right position
210:       counter = 0 
211:       while solutionIterator.hasNext
212:         solution = solutionIterator.next
213:         
214:         temparray = []
215:         # get the value associated with a variable in this specific solution
216:         temparray[0] = convertSesame2ActiveRDF(solution.getValue(variables[0]))
217:         results[counter] = temparray
218:         counter = counter + 1
219:       end    
220:     else
221:     # if there is more then one variable the results array looks like this: 
222:     # results = [ [Value From First Solution For First Variable, Value From First Solution For Second Variable, ...],
223:     #             [Value From Second Solution For First Variable, Value From Second Solution for Second Variable, ...], ...]
224:       counter = 0
225:       while solutionIterator.hasNext
226:         solution = solutionIterator.next
227:         
228:         temparray = []
229:         for n in 1..sizeOfVariables
230:           value = convertSesame2ActiveRDF(solution.getValue(variables[n-1]))
231:           temparray[n-1] = value
232:         end   
233:         results[counter] = temparray
234:         counter = counter + 1       
235:       end    
236:     end
